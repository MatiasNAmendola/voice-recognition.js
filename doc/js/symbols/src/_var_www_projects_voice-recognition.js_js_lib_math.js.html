  5 (function() {
  6 	if (!window.Utils) { //Define utils' namespace
  7 		window.Utils = {};
  8 	}
  9 
 10 	if (Utils.Math) { //If the library is already loaded
 11 		return;
 12 	}
 13 
 14 	/**
 15 	 * Math utils namespace.
 16 	 * @type {Object}
 17 	 * @namespace
 18 	 */
 19 	Utils.Math = {};
 20 
 21 	/**
 22 	 * Round a number to a specified number of decimals.
 23 	 * @param  {Number} num        The number.
 24 	 * @param  {Number} [numOfDec] The number of decimals.
 25 	 * @return {Number}            The rounded number.
 26 	 */
 27 	Utils.Math.getNumWithSetDec = function getNumWithSetDec(num, numOfDec) {
 28 		numOfDec = (typeof numOfDec == 'number') ? numOfDec : Utils.Options.get('utils.math.precision');
 29 
 30 		var pow10s = Math.pow( 10, numOfDec );
 31 		return ( numOfDec ) ? Math.round( pow10s * num ) / pow10s : num;
 32 	};
 33 
 34 	/**
 35 	 * Get the average from an array of numbers.
 36 	 * @param  {Number[]} numArr   The array of numbers.
 37 	 * @param  {Number} [numOfDec] The number of decimals.
 38 	 * @param  {Number[]} [range]  The index range.
 39 	 * @return {Number}            The average.
 40 	 */
 41 	Utils.Math.getAverageFromNumArr = function getAverageFromNumArr(numArr, numOfDec, range) {
 42 		range = range || [0, numArr.length - 1];
 43 
 44 		if (numArr.length == 0) {
 45 			return Number.NaN;
 46 		}
 47 
 48 		var i = range[1] + 1,
 49 			sum = 0;
 50 		while( i-- && i >= range[0] ){
 51 			sum += numArr[ i ];
 52 		}
 53 
 54 		return Utils.Math.getNumWithSetDec( (sum / (range[1] - range[0] + 1)), numOfDec );
 55 	};
 56 
 57 	/**
 58 	 * Get the variance from an array of numbers.
 59 	 * @param  {Number[]} numArr   The array of numbers.
 60 	 * @param  {Number} [numOfDec] The number of decimals.
 61 	 * @param  {Number[]} [range]  The index range.
 62 	 * @return {Number}            The variance.
 63 	 */
 64 	Utils.Math.getVariance = function getVariance(numArr, numOfDec, range) {
 65 		range = range || [0, numArr.length - 1];
 66 
 67 		if (numArr.length == 0) {
 68 			return Number.NaN;
 69 		}
 70 
 71 		var avg = Utils.Math.getAverageFromNumArr(numArr, numOfDec, range), 
 72 			i = range[1] + 1,
 73 			v = 0;
 74 	 
 75 		while( i-- && i >= range[0] ){
 76 			v += Math.pow( (numArr[ i ] - avg), 2 );
 77 		}
 78 		v /= range[1] - range[0] + 1;
 79 		return Utils.Math.getNumWithSetDec(v, numOfDec);
 80 	};
 81 
 82 	/**
 83 	 * Get the standard deviation from an array of numbers.
 84 	 * @param  {Number[]} numArr     The array of numbers.
 85 	 * @param  {Number}   [numOfDec] The number of decimals.
 86 	 * @param  {Number[]  [range]}   The index range.
 87 	 * @return {Number}              The standard deviation.
 88 	 */
 89 	Utils.Math.getStandardDeviation = function getStandardDeviation(numArr, numOfDec, range) {
 90 		var stdDev = Math.sqrt(Utils.Math.getVariance(numArr, numOfDec, range));
 91 		return Utils.Math.getNumWithSetDec(stdDev, numOfDec);
 92 	};
 93 })();
