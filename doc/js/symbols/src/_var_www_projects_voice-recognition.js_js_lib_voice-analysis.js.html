1 /**
  2  * A Fast Fourrier Transform.
  3  * @see <a href="https://github.com/corbanbrook/dsp.js">https://github.com/corbanbrook/dsp.js</a>
  4  * @constructor
  5  * @param {Number} bufferSize The buffer size (e.g. 1024).
  6  * @param {Number} sampleRate The sample rate (e.g. 44100).
  7  */
  8 var FFT = function FFT(bufferSize, sampleRate) {
  9 	this.bufferSize   = bufferSize;
 10 	this.sampleRate   = sampleRate;
 11 	this.spectrum     = new Float32Array(bufferSize/2);
 12 	this.real         = new Float32Array(bufferSize);
 13 	this.imag         = new Float32Array(bufferSize);
 14 	this.reverseTable = new Uint32Array(bufferSize);
 15 	this.sinTable     = new Float32Array(bufferSize);
 16 	this.cosTable     = new Float32Array(bufferSize);
 17 
 18 	var limit = 1,
 19 	bit = bufferSize >> 1;
 20 
 21 	while ( limit < bufferSize ) {
 22 		for ( var i = 0; i < limit; i++ ) {
 23 			this.reverseTable[i + limit] = this.reverseTable[i] + bit;
 24 		}
 25 
 26 		limit = limit << 1;
 27 		bit = bit >> 1;
 28 	}
 29 
 30 	for ( var i = 0; i < bufferSize; i++ ) {
 31 		this.sinTable[i] = Math.sin(-Math.PI/i);
 32 		this.cosTable[i] = Math.cos(-Math.PI/i);
 33 	}
 34 };

 36 /**
 37  * Forward a signal.
 38  * @param  {Float32Array} buffer The audio data.
 39  */
 40 FFT.prototype.forward = function forward(buffer) {
 41 	var bufferSize   = this.bufferSize,
 42 	cosTable     = this.cosTable,
 43 	sinTable     = this.sinTable,
 44 	reverseTable = this.reverseTable,
 45 	real         = this.real,
 46 	imag         = this.imag,
 47 	spectrum     = this.spectrum;
 48 
 49 	if ( bufferSize !== buffer.length ) {
 50 		throw "Supplied buffer is not the same size as defined FFT. FFT Size: " + bufferSize + " Buffer Size: " + buffer.length;
 51 	}
 52 
 53 	for ( var i = 0; i < bufferSize; i++ ) {
 54 		real[i] = buffer[reverseTable[i]];
 55 		imag[i] = 0;
 56 	}
 57 
 58 	var halfSize = 1,
 59 	phaseShiftStepReal,	
 60 	phaseShiftStepImag,
 61 	currentPhaseShiftReal,
 62 	currentPhaseShiftImag,
 63 	off,
 64 	tr,
 65 	ti,
 66 	tmpReal,	
 67 	i;
 68 
 69 	while ( halfSize < bufferSize ) {
 70 		phaseShiftStepReal = cosTable[halfSize];
 71 		phaseShiftStepImag = sinTable[halfSize];
 72 		currentPhaseShiftReal = 1.0;
 73 		currentPhaseShiftImag = 0.0;
 74 
 75 		for ( var fftStep = 0; fftStep < halfSize; fftStep++ ) {
 76 			i = fftStep;
 77 
 78 			while ( i < bufferSize ) {
 79 				off = i + halfSize;
 80 				tr = (currentPhaseShiftReal * real[off]) - (currentPhaseShiftImag * imag[off]);
 81 				ti = (currentPhaseShiftReal * imag[off]) + (currentPhaseShiftImag * real[off]);
 82 
 83 				real[off] = real[i] - tr;
 84 				imag[off] = imag[i] - ti;
 85 				real[i] += tr;
 86 				imag[i] += ti;
 87 
 88 				i += halfSize << 1;
 89 			}
 90 
 91 			tmpReal = currentPhaseShiftReal;
 92 			currentPhaseShiftReal = (tmpReal * phaseShiftStepReal) - (currentPhaseShiftImag * phaseShiftStepImag);
 93 			currentPhaseShiftImag = (tmpReal * phaseShiftStepImag) + (currentPhaseShiftImag * phaseShiftStepReal);
 94 		}
 95 
 96 		halfSize = halfSize << 1;
 97 	}
 98 
 99 	i = bufferSize/2;
100 	while(i--) {
101 		spectrum[i] = 2 * Math.sqrt(real[i] * real[i] + imag[i] * imag[i]) / bufferSize;
102 	}
103 };

168 	 * Get/set frequencies on which this analysis is made.
169 	 * @param  {Number|Number[]} freq Frequencies.
170 	 * @return {Number[]|null}        Frequencies on which this analysis is made.
171 	 */
172 	frequencies: function frequencies(freq) {
173 		if (typeof freq == 'undefined') {
174 			return this._frequencies;
175 		} else {
176 			switch (typeof freq) {
177 				case 'number':
178 					freq = [freq];
179 					break;
180 				case 'string':
181 					if (/^[0-9]+-[0-9]+$/.test(freq)) {
182 						var result = /^([0-9]+)-([0-9]+)$/.exec(freq),
183 						range = [parseInt(result[1]), parseInt(result[2])];
184 
185 						if (range[0] > range[1]) {
186 							range = [range[1], range[0]];
187 						}
188 
189 						freq = [];
190 						for (var i = range[0]; i <= range[1]; i++) {
191 							freq.push(i);
192 						}
193 					} else if (/^[0-9]+$/.test(freq)) {
194 						freq = [parseInt(freq)];
195 					}
196 					break;
197 				case 'object':
198 					if (freq instanceof Array) {
199 						//Use the array as is
200 						//freq = freq;
201 					} else {
202 						return false;
203 					}
204 					break;
205 				default:
206 					return false;
207 			}
208 
209 			freq.sort();
210 
211 			this._frequencies = freq;
212 		}

313 	/**
314 	 * Method to call when audio data is available.
315 	 * @param  {Float32Array} fb The audio data (frame buffer).
316 	 * @param  {Number} t  The audio time.
317 	 */
318 	audioAvailable: function audioAvailable(fb, t) {
319 		if (this.status() < 1) { //If it's a new analysis, reset this one
320 			this.reset();
321 		}
322 
323 		if (this._startTime === null) { //If it's the first one this method is called, register actual time as start time
324 			this._startTime = new Date().getTime() / 1000;
325 		}
326 		if (typeof t != 'number') { //If no time is specified, let's deduce it from the start time
327 			t = (new Date).getTime() / 1000 - this._startTime;
328 		}
329 
330 		//Forward signal to FFT
331 		var signal = new Float32Array(fb.length / this._channels);
332 
333 		for (var i = 0, fbl = this._frameBufferLength / 2; i < fbl; i++ ) {
334 			// Assuming interlaced stereo channels,
335 			// need to split and merge into a stero-mix mono signal
336 			signal[i] = (fb[2*i] + fb[2*i+1]) / 2;
337 		}
338 
339 		this._fft.forward(signal); //Forward the signal
340 
341 		//Variables for canvas drawing
342 		var showFFT = Utils.Options.get('voice.comparing.showFFT'), // Do we have to draw the FFT on the canvas ?
343 		canvas = this.control('canvas')[0],
344 		ctx = canvas.getContext('2d');
345 
346 		if (showFFT) { //If FFT is shown, clear the canvas
347 			ctx.clearRect(0,0, canvas.width, canvas.height);
348 		}
349 
350 		//Get the selected magnitudes
351 		var analysisFreq = this.frequencies(); //Frequencies on which this analysis is made
352 		var magnitudes = new Float32Array(analysisFreq.length);
353 
354 		var magnitude,
355 		freqIndex = 0,
356 		maxMagnitude = 0,
357 		maxMagnitudeFreq;
358 
359 		for (var i = 0; i < this._fft.spectrum.length; i++ ) {
360 			var isMagnitudeSaved = ($.inArray(i, analysisFreq) != -1); //Is this magnitude saved ?
361 
362 			if (!showFFT && !isMagnitudeSaved) {
363 				continue; //Ignore this magnitude
364 			}
365 
366 			//Multiply spectrum by a zoom value
367 			magnitude = this._fft.spectrum[i] * 4000;
368 
369 			if (showFFT) {
370 				//Draw rectangle bars for each frequency bin
371 				ctx.fillRect(i * 4, canvas.height, 3, - magnitude);
372 			}
373 
374 			if (isMagnitudeSaved) { //If we have to save this magnitude
375 				magnitudes[freqIndex] = magnitude;
376 				freqIndex++;
377 
378 				if (magnitude > maxMagnitude) { //Is this the max. magnitude ?
379 					maxMagnitude = magnitude;
380 					maxMagnitudeFreq = freqIndex;
381 				}
382 			}
383 		}
384 
385 		if (this._magnitudes.length < this._maxAnalysisLength) { //If this analysis is not overflowed
386 			//Save data in arrays
387 			this._magnitudes.push(magnitudes);
388 			this._time.push(t);
389 		}
390 
391 		if (this._maxMagnitude < maxMagnitude) { //Analysis' max magnitude
392 			this._maxMagnitude = maxMagnitude;
393 			this._maxMagnitudeFreq = maxMagnitudeFreq;
394 		}
395 	},

402 	/**
403 	 * Process the audio data.
404 	 */
405 	processData: function processData() {
406 		if (this.status() < 2 || !this._magnitudes) {
407 			return false;
408 		}
409 
410 		this.notify('start');
411 
412 		//Let's standardize magnitudes
413 		Utils.logMessage('---------');
414 		Utils.logMessage('Standardizing magnitudes...');
415 		Utils.logMessage('Max magnitude for this analysis : ' + this._maxMagnitude);
416 
417 		this._standardizedMagnitudes = [];
418 		var i, j, magnitudes;
419 		for (i = 0; i < this._magnitudes.length; i++) { //For each data frame
420 			magnitudes = this._magnitudes[i]; //Get magnitudes saved at this time
421 
422 			this._standardizedMagnitudes[i] = new Float32Array(magnitudes.length); //Create a new array to store standardized data
423 
424 			for (j = 0; j < magnitudes.length; j++) {
425 				//We want the magnitude in % of the max magnitude
426 				this._standardizedMagnitudes[i][j] = Utils.Math.getNumWithSetDec(magnitudes[j] / this._maxMagnitude * 100);
427 			}
428 		}
429 
430 		//Now we can determine when does the voice begin and end
431 		//Get the tolerance & precision options
432 		var tolerance = Utils.Options.get('voice.analysis.tolerance'),
433 		precision = Utils.Options.get('voice.analysis.precision');
434 
435 		this._range = [0, this._dataIndex]; //The voice range
436 
437 		//Internal options
438 		var threshold = 8, //Required freq. magnitude in % to determine the begining/end of the speech
439 		requiredFollowingPts = 4; //Required following points wich are > the threshold to determine the begining/end of the speech
440 
441 		Utils.logMessage('---------');
442 		Utils.logMessage('Determining the beginning...');
443 
444 		var startedSince = 0, magnitude;
445 		for (var i = 0; i < this._magnitudes.length; i++) {
446 			magnitude = this._standardizedMagnitudes[i][0];
447 
448 			if (magnitude > threshold) {
449 				Utils.logMessage('Peak detected', i, magnitude + ' > ' + threshold);
450 				startedSince++;
451 			} else if (startedSince) {
452 				startedSince = 0;
453 			}
454 
455 			if (startedSince >= requiredFollowingPts) {
456 				this._range[0] = i - startedSince + 1;
457 				Utils.logMessage('=> Begining detected', this._range[0]);
458 				break;
459 			}
460 		}
461 
462 		Utils.logMessage('---------');
463 		Utils.logMessage('Determining the end...');
464 
465 		var endedSince = 0, magnitude;
466 		for (var i = this._magnitudes.length - 1; i >= 0; i--) {
467 			magnitude = this._standardizedMagnitudes[i][0];
468 
469 			if (magnitude > threshold) {
470 				Utils.logMessage('Peak detected', i, magnitude + ' > ' + threshold);
471 				endedSince++;
472 			} else if (endedSince) {
473 				endedSince = 0;
474 			}
475 
476 			if (endedSince >= requiredFollowingPts) {
477 				this._range[1] = i + endedSince - 1;
478 				Utils.logMessage('=> End detected', this._range[1]);
479 				break;
480 			}
481 		}
482 
483 		//And finally standardize time
484 		Utils.logMessage('---------');
485 		Utils.logMessage('Standardizing time...');
486 
487 		this._standardizedTime = [];
488 
489 		var startTime = this._time[this._range[0]], //Time when the voice begins
490 		endTime = this._time[this._range[1]], //Time when the voice ends
491 		duration = endTime - startTime; //Voice duration
492 
493 		//Display some stats
494 		Utils.logMessage('Number of points collected : ' + (this._range[1] - this._range[0] + 1));
495 		Utils.logMessage('Speaking duration : ' + duration + ' ('+startTime+' -> '+endTime+')');
496 
497 		var t;
498 		for (var i = 0; i < this._magnitudes.length; i++) {
499 			t = this._time[i] - startTime; //Time since the begining
500 
501 			//We want the time in %
502 			this._standardizedTime[i] = Utils.Math.getNumWithSetDec(t / duration * 100);
503 		}
504 
505 
506 		this._updateStatus(3);
507 
508 		this.notify('complete');
509 	},
510 	/**
511 	 * Export the audio data.
512 	 * @param  {String} format The exporting format : csv or json.
513 	 */
514 	exportData: function exportData(format) {
515 		format = format || 'json';
516 
517 		switch (format) { //Different methods for deffierent export formats
518 			case 'csv': //CSV : for spreadsheet
519 				var out = 'Time;Standardized time;Frequency;Magnitude;Standardized magnitude',
520 				status = 0,
521 				analysisFreq = this.frequencies(); //Frequencies on which this analysis is made
522 
523 				var i, j, magnitudes;
524 				for (i = 0; i < this._magnitudes.length; i++) {
525 					magnitudes = this._magnitudes[i]; //Get magnitudes saved at this time
526 
527 					for (j = 0; j < magnitudes.length; j++) {
528 						out += "\n"+this._time[i]+';'+this._standardizedTime[i]+';'+analysisFreq[j]+';'+magnitudes[j]+';'+this._standardizedMagnitudes[i][j];
529 					}
530 				}
531 
532 				Utils.Export.exportCSV(out);
533 				break;
534 			case 'json': //JSON : to save as a model
535 				var dataToExport = {};
536 
537 				for (var i = this._range[0]; i <= this._range[1]; i++) {
538 					dataToExport[this._standardizedTime[i]] = this._standardizedMagnitudes[i];
539 				}
540 
541 				var i, j, magnitudes;
542 				for (i = 0; i < this._magnitudes.length; i++) {
543 					magnitudes = this._magnitudes[i]; //Get magnitudes saved at this time
544 
545 					dataToExport[this._standardizedTime[i]] = magnitudes;
546 				}
547 
548 				Utils.Export.exportJSON(dataToExport);
549 				break;
550 		}
551 	}
