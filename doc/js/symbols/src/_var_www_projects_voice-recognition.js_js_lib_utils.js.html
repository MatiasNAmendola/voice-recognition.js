1 /**
  2  * @fileOverview Various utils.
  3  * @author $imon
  4  */
  5 
  6 (function() {
  7 	if (window.Utils) { //Define utils' namespace
  8 		return;
  9 	}
 10 
 11 	var Utils;
 12 	
 13 	/**
 14 	 * Some various utils.
 15 	 * @type {Object}
 16 	 * @namespace
 17 	 */
 18 	Utils = {};
 19 
 20 	/**
 21 	 * Log a message in the console.
 22 	 * @function
 23 	 */
 24 	Utils.logMessage = function logMessage() {
 25 		if (Utils.Options.get('utils.logMessages')) {
 26 			console.log.apply(window, Array.prototype.slice.call(arguments));
 27 		}
 28 	};
 29 
 30 
 31 	/**
 32 	 * Tool to export data.
 33 	 * @type {Object}
 34 	 * @namespace
 35 	 */
 36 	Utils.Export = {};
 37 
 38 	/**
 39 	 * Export data.
 40 	 * @param  {String} contents The raw data.
 41 	 * @param  {String} type     The MIME type of the data.
 42 	 */
 43 	Utils.Export.exportData = function exportData(contents, type) {
 44 		window.open('data:'+type+';base64,'+window.btoa(contents));
 45 	};
 46 
 47 	/**
 48 	 * Export data to CSV.
 49 	 * @param  {String} csv The CSV data.
 50 	 */
 51 	Utils.Export.exportCSV = function exportCSV(csv) {
 52 		return Utils.Export.exportData(csv.replace(/\./g,','), 'text/csv');
 53 	};
 54 
 55 	/**
 56 	 * Export data to JSON.
 57 	 * @param  {Object} data The data.
 58 	 */
 59 	Utils.Export.exportJSON = function exportJSON(data) {
 60 		return Utils.Export.exportData(JSON.stringify(data, null, "\t"), 'application/json');
 61 	};
 62 
 63 	window.Utils = Utils; //Export API
 64 })();
