1 /**
  2  * @fileOverview Operation's progress management.
  3  * @author $imon
  4  */
  5 
  6 (function() {
  7 	if (!window.Utils) { //Define utils' namespace
  8 		window.Utils = {};
  9 	}
 10 
 11 	if (Utils.Progress) { //If the library is already loaded
 12 		return;
 13 	}
 14 
 15 	/**
 16 	 * An operation's progress data.
 17 	 * @constructor
 18 	 * @param {Object} options Progress' options.
 19 	 * @param {Number} options.part Number of parts in the progress.
 20 	 */
 21 	Utils.Progress = function Progress(options) {
 22 		Utils.Observable.call(this); //Inheritance from Observable
 23 
 24 		this._value = 0;
 25 		this._message = '';
 26 		this._error = false;
 27 
 28 		options = options || {};
 29 
 30 		this._parts = options.parts || 1;
 31 	};
 32 	Utils.Progress.prototype = {
 33 		/**
 34 		 * Get/set the progress' value.
 35 		 * @param  {Number} [value] The new value, in %.
 36 		 * @return {Number}         The value.
 37 		 */
 38 		value: function value(value) {
 39 			if (typeof value == 'undefined') {
 40 				return this._value;
 41 			} else {
 42 				value = parseInt(value);
 43 
 44 				if (isNaN(value)) {
 45 					value = 0;
 46 				}
 47 				if (value < 0) {
 48 					value = 0;
 49 				}
 50 				if (value > 100) {
 51 					value = 100;
 52 				}
 53 
 54 				this._value = value;
 55 
 56 				this.notify('update', { value: value, message: this.message(), error: this.error() });
 57 			}
 58 		},
 59 		/**
 60 		 * Get/set number of parts.
 61 		 * @param  {Number} value The number of parts.
 62 		 * @return {Number}       The number of parts.
 63 		 */
 64 		parts: function parts(value) {
 65 			if (typeof value == 'undefined') {
 66 				return this._parts;
 67 			} else {
 68 				value = parseInt(value);
 69 
 70 				if (isNaN(value)) {
 71 					value = 1;
 72 				}
 73 
 74 				this._parts = value;
 75 			}
 76 		},
 77 		/**
 78 		 * Mark a part as completed.
 79 		 * @param  {Number} [factor=1] The part's importance.
 80 		 */
 81 		partComplete: function partComplete(factor) {
 82 			factor = (typeof factor == 'number') ? factor : 1;
 83 
 84 			var percentage = this.value() + 1 / this.parts() * 100 * factor;
 85 			this.value(percentage);
 86 		},
 87 		/**
 88 		 * Get/set the progress' message.
 89 		 * @param  {String} msg The message.
 90 		 * @return {String}     The message.
 91 		 */
 92 		message: function message(msg) {
 93 			if (typeof msg == 'undefined') {
 94 				return this._message;
 95 			} else {
 96 				msg = String(msg);
 97 
 98 				this._message = msg;
 99 
100 				this.notify('update', { value: this.value(), message: msg, error: this.error() });
101 			}
102 		},
103 		/**
104 		 * Determine if the operation has failed / mark the operation as failed.
105 		 * @param  {String} msg A message to describe the error.
106 		 * @return {Boolean}    True if the operation has failed.
107 		 */
108 		error: function error(msg) {
109 			if (typeof msg == 'undefined') {
110 				return this._error;
111 			} else {
112 				if (msg === false) {
113 					this._error = false;
114 
115 					this.notify('update', { value: this.value(), message: this.message(), error: false });
116 				} else {
117 					this._error = true;
118 
119 					this.message(msg);
120 				}
121 			}
122 		},
123 		/**
124 		 * Reset the progress.
125 		 */
126 		reset: function reset() {
127 			this.error(false);
128 			this.value(0);
129 		}
130 	};
131 
132 	Utils.inherit(Utils.Progress, Utils.Observable); //Inheritance from Observable
133 })();
