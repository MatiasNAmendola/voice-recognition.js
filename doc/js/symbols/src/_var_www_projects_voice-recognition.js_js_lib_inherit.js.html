  6 (function() {
  7 	if (!window.Utils) { //Define utils' namespace
  8 		window.Utils = {};
  9 	}
 10 
 11 	if (Utils.inherit) { //If the library is already loaded
 12 		return;
 13 	}
 14 
 15 	/**
 16 	 * An implementation of inheritance.
 17 	 * @function
 18 	 * @param {Object} C The child class.
 19 	 * @param {Object} P The parent class.
 20 	 */
 21 	Utils.inherit = function inherit(C, P) {
 22 		var F = function() {};
 23 		F.prototype = P.prototype;
 24 		C.prototype = $.extend({}, new F(), C.prototype);
 25 		C.uber = P.prototype;
 26 		C._parent = P;
 27 		C.prototype.constructor = C;
 28 	};
 29 
 30 	/**
 31 	 * Specify if an object is an instance of a class.
 32 	 * @function
 33 	 * @param {Object} instance The object to test.
 34 	 * @param {Object} obj The class.
 35 	 * @returns {Boolean} True if the object if an instance of the specified class.
 36 	 */
 37 	Utils.isInstanceOf = function isInstanceOf(instance, obj) {
 38 		if (!instance || typeof instance != 'object' || !obj) {
 39 			return false;
 40 		}
 41 
 42 		var current;
 43 		do {
 44 			if (current) {
 45 				current = current._parent;
 46 			} else {
 47 				current = instance.constructor;
 48 			}
 49 			
 50 			if (current === obj) {
 51 				return true;
 52 			}
 53 		} while (current._parent);
 54 
 55 		try {
 56 			if (instance instanceof obj) {
 57 				return true;
 58 			}
 59 		} catch(e) {}
 60 		
 61 		return false;
 62 	};
 63 })();
